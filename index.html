<script>

/* ---------- 3-D ROTATION & ZOOM ---------- */

const model = document.getElementById('model');

let rotX = -20, rotY = 0, distance = 1;

let isDragging = false, startX, startY;


function updateTransform(){

  model.style.transform = `translate(-50%,-50%) rotateX(${rotX}deg) rotateY(${rotY}deg) scale(${distance})`;

}

updateTransform();


document.addEventListener('mousedown', e=>{ isDragging=true; startX=e.clientX; startY=e.clientY; });

document.addEventListener('mousemove', e=>{

  if(!isDragging) return;

  const dx = e.clientX - startX;

  const dy = e.clientY - startY;

  rotY += dx * 0.5;

  rotX -= dy * 0.5;

  startX = e.clientX; startY = e.clientY;

  updateTransform();

});

document.addEventListener('mouseup', ()=>{ isDragging=false; });

document.addEventListener('wheel', e=>{

  distance = Math.max(0.5, Math.min(3, distance - e.deltaY*0.001));

  updateTransform();

});

/* touch support */

let touchStartX, touchStartY, touchStartDist;

document.addEventListener('touchstart', e=>{

  if(e.touches.length===1){

    isDragging=true; touchStartX=e.touches[0].clientX; touchStartY=e.touches[0].clientY;

  }else if(e.touches.length===2){

    const dx = e.touches[0].clientX - e.touches[1].clientX;

    const dy = e.touches[0].clientY - e.touches[1].clientY;

    touchStartDist = Math.hypot(dx,dy);

  }

});

document.addEventListener('touchmove', e=>{

  if(e.touches.length===1 && isDragging){

    const dx = e.touches[0].clientX - touchStartX;

    const dy = e.touches[0].clientY - touchStartY;

    rotY += dx * 0.5;

    rotX -= dy * 0.5;

    touchStartX = e.touches[0].clientX;

    touchStartY = e.touches[0].clientY;

    updateTransform();

  }else if(e.touches.length===2){

    const dx = e.touches[0].clientX - e.touches[1].clientX;

    const dy = e.touches[0].clientY - e.touches[1].clientY;

    const curDist = Math.hypot(dx,dy);

    distance *= curDist / touchStartDist;

    distance = Math.max(0.5, Math.min(3, distance));

    touchStartDist = curDist;

    updateTransform();

  }

});

document.addEventListener('touchend', ()=>{ isDragging=false; });


/* ---------- HOT-SPOTS & TOOLTIP ---------- */

const tooltip = document.getElementById('tooltip');

document.querySelectorAll('.hotspot').forEach(dot=>{

  dot.addEventListener('click',()=>{

    tooltip.textContent = dot.dataset.msg;

    tooltip.style.opacity = 1;

    clearTimeout(tooltip.hideTimer);

    tooltip.hideTimer = setTimeout(()=>{ tooltip.style.opacity=0; }, 4500);

  });

});


/* ---------- OPTIONAL: keep hot-spots stuck to 3-D points (simple projection) ---------- */

function project3D(x,y,z){

  const scale = 150; /* half-size of cube */

  const sx = x*scale + innerWidth/2;

  const sy = -y*scale + innerHeight/2;

  return {x:sx, y:sy};

}

const hotData = [

  {x: 0.4, y: 0.6, z: 0.9}, // damage tip

  {x:-0.4, y: 0.6, z: 0.9}, // damage tip opposite

  {x: 0  , y: 0  , z: 0  }, // pivot

  {x: 0.2, y:-0.2, z: 0.7}, // hinge germ

  {x:-0.2, y:-0.2, z: 0.7}  // groove germ

];

const dots = document.querySelectorAll('.hotspot');

function updateHotspots(){

  const sinX = Math.sin(rotX*Math.PI/180), cosX = Math.cos(rotX*Math.PI/180);

  const sinY = Math.sin(rotY*Math.PI/180), cosY = Math.cos(rotY*Math.PI/180);

  hotData.forEach((p,i)=>{

    // simple rotation around Y then X

    const ry = cosY * p.x + sinY * p.z;

    const rz = -sinY * p.x + cosY * p.z;

    const rx = cosX * p.y + sinX * rz;

    const rz2 = -sinX * p.y + cosX * rz;

    const proj = project3D(ry, rx, rz2);

    dots[i].style.left = proj.x + 'px';

    dots[i].style.top  = proj.y + 'px';

    // hide if behind

    dots[i].style.opacity = (rz2 > -0.3) ? 1 : 0.3;

  });

}

function loop(){

  updateHotspots();

  requestAnimationFrame(loop);

}

loop();

</script>
